Class ZombieScientistHammer : Actor
{
	Default
	{
		//$Category Monsters/ZombieScientists
		//$Title "Hammer"
		obituary "%o was bludgeoned to death by a zombie scientist.";
		health 20;
		mass 90;
		speed 16;
		Radius 20;
		Height 52;
		painchance 200;
		seesound "zombiescientistM/sight";
		painsound "zombiescientistM/pain";
		deathsound "zombiescientistM/death";
		activesound "zombiescientistM/active";
		MONSTER;
		+FLOORCLIP;
		+WINDTHRUST;
	}
	States
	{
	Spawn:
		SCZH AB 10 A_Look();
		loop;
	See:
		SCZH AABBCCDD 4 A_Chase();
		loop;
	Melee:
		SCZH E 8 A_FaceTarget();
		SCZH F 16 A_CustomMeleeAttack(random (2, 6) *5, "zombiescientist/blunthit", "zombiescientist/swing");
		SCZH E 4;
		goto See;
	Pain:
		SCZH G 3;
		SCZH G 3 A_Pain();
		goto See;
	Death:
		SCZH H 5;
		SCZH I 5 A_Scream();
		SCZH J 5 A_NoBlocking();
		SCZH K 5;
		SCZH L 5;
		SCZH M 5;
		SCZH N -1;
		stop;
	XDeath:
		SCZH O 5;
		SCZH P 5 A_XScream();
		SCZH Q 5 A_NoBlocking();
		SCZH RSTUV 5;
		SCZH W -1;
		stop;
	Raise:
		SCZH MLKJIH 5;
		goto See;
	}
}
