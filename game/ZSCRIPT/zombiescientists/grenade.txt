Class ZombieScientistGrenade : Actor
{
	Default
	{
		//$Category Monsters/ZombieScientists
		//$Title "Grenade"
		obituary "%o was fragged by a zombie scientist.";
		health 18;
		mass 70;
		speed 10;
		Radius 19;
		Height 52;
		painchance 200;
		MaxTargetRange 512;
		seesound "zombiescientistF/sight";
		painsound "zombiescientistF/pain";
		deathsound "zombiescientistF/death";
		activesound "zombiescientistF/active";
		MONSTER;
		+FLOORCLIP;
		+AVOIDMELEE;
		+WINDTHRUST;
	}
	States
	{
	Spawn:
		FSZG AB 10 A_Look();
		loop;
	See:
		FSZG AABBCCDD 4 A_Chase();
		loop;
	Missile:
		FSZG E 12 A_FaceTarget();
		FSZG F 10
		{
			A_StartSound("zombiescientist/swing");
			A_SpawnProjectile("ZombieScientistGrenadeProjectile",32,8,random(-8,8),CMF_OFFSETPITCH,random(-20,-10));
		}
		goto See;
	Pain:
		FSZG G 3;
		FSZG G 3 A_Pain();
		goto See;
	Death:
		FSZG H 5 A_SpawnProjectile("ZombieScientistGrenadeProjectileDeath",32,8,random(0,359),CMF_OFFSETPITCH,random(-90,-45));
		FSZG I 5 A_Scream();
		FSZG J 5 A_NoBlocking();
		FSZG K 5;
		FSZG L 5;
		FSZG M 5;
		FSZG N -1;
		stop;
	XDeath:
		FSZG O 5 A_SpawnProjectile("ZombieScientistGrenadeProjectileDeath",32,8,random(0,359),CMF_OFFSETPITCH,random(-90,-45));
		FSZG P 5 A_XScream();
		FSZG Q 5 A_NoBlocking();
		FSZG RSTUV 5;
		FSZG W -1;
		stop;
	Raise:
		FSZG MLKJIH 5;
		goto See;
	}
}

Class ZombieScientistGrenadeProjectile : Actor
{
	int rolldir;
	Default
	{
		Radius 5;
		Height 5;
		Scale 0.4;
		Speed 13;
		BounceType "Doom";
		BounceFactor 0.8;
		WallBounceFactor 0.5;
		BounceCount 6;
		BounceSound "zombiescientist/grenbounce";
		DeathSound "zombiescientist/grenexplode";
		Decal "RevenantScorch";
		Projectile;
		-NOGRAVITY;
		+ROLLSPRITE;
		+ROLLCENTER;
	}
	States
	{
	Spawn:
		FSZG X 1 Nodelay
		{
			rolldir = randompick(-10,10);
			roll = 45;
		}
	See:
		FSZG X 1
		{
			roll += rolldir;
		}
		Loop;
	Death:
		EXP4 A 1 bright
		{
			roll = 0;
			A_SetScale(1,1);
			A_Explode(128,128);
			bNOGRAVITY = true;
		}
		EXP4 BCDEFGHIJKLMNOPQRSTUVWXYZ 1 bright;
		Stop;
	}
}

Class ZombieScientistGrenadeProjectileDeath : ZombieScientistGrenadeProjectile
{
	Default
	{
		Speed 3;
	}
}
