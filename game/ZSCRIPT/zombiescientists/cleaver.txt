Class ZombieScientistCleaver : Actor
{
	Default
	{
		//$Category Monsters/ZombieScientists
		//$Title "Cleaver"
		obituary "%o was butchered by a zombie scientist.";
		health 18;
		mass 70;
		speed 20;
		Radius 19;
		Height 52;
		painchance 200;
		seesound "zombiescientistF/sight";
		painsound "zombiescientistF/pain";
		deathsound "zombiescientistF/death";
		activesound "zombiescientistF/active";
		MONSTER;
		+FLOORCLIP;
		+WINDTHRUST;
	}
	States
	{
	Spawn:
		FSZC AB 10 A_Look();
		loop;
	See:
		FSZC AABBCCDD 4 A_Chase();
		loop;
	Melee:
		FSZC E 4 A_FaceTarget();
		FSZC F 8 A_CustomMeleeAttack(random (1, 5) *5, "zombiescientist/sharphit", "zombiescientist/swing");
		FSZC D 4;
		goto See;
	Pain:
		FSZC G 3;
		FSZC G 3 A_Pain();
		goto See;
	Death:
		FSZC H 5;
		FSZC I 5 A_Scream();
		FSZC J 5 A_NoBlocking();
		FSZC K 5;
		FSZC L 5;
		FSZC M 5;
		FSZC N -1;
		stop;
	XDeath:
		FSZC O 5;
		FSZC P 5 A_XScream();
		FSZC Q 5 A_NoBlocking();
		FSZC RSTUV 5;
		FSZC W -1;
		stop;
	Raise:
		FSZC MLKJIH 5;
		goto See;
	}
}
