Class ZombieScientistFreeze : Actor
{
	Default
	{
		//$Category Monsters/ZombieScientists
		//$Title "Freezerifle"
		obituary "%o was frozen by a zombie scientist.";
		health 20;
		mass 100;
		speed 8;
		Radius 20;
		Height 52;
		painchance 200;
		seesound "zombiescientistM/sight";
		painsound "zombiescientistM/pain";
		deathsound "zombiescientistM/death";
		activesound "zombiescientistM/active";
		dropitem "Cell";
		MONSTER;
		+FLOORCLIP;
		+AVOIDMELEE;
		MissileChanceMult 0.125;
		+WINDTHRUST;
	}
	States
	{
	Spawn:
		SZFR AB 10 A_Look();
		loop;
	See:
		SZFR AABBCCDD 4 A_Chase();
		loop;
	Missile:
		SZFR E 6 A_FaceTarget();
		SZFR F 5 A_SpawnProjectile("ZombieScientistFreezeBlast",32,7,Random(-3,3),CMF_OFFSETPITCH,frandom(-0.3,0.3));
		SZFR E 12;
	goto See;
		Pain:
		SZFR G 3;
		SZFR G 3 A_Pain();
		goto See;
	Death:
		SZFR H 5 A_SpawnItemEx("ZombieScientistFreezeDeathExplosion", 0.0, 0.0, 32.0, 0.0, 0.0, 0.0, 0.0, SXF_NOCHECKPOSITION, 0);
		SZFR I 5 A_Scream();
		SZFR J 5 A_NoBlocking();
		SZFR K 5;
		SZFR L 5;
		SZFR M 5;
		SZFR N -1;
		stop;
	XDeath:
		SZFR O 5 A_SpawnItemEx("ZombieScientistFreezeDeathExplosion", 0.0, 0.0, 32.0, 0.0, 0.0, 0.0, 0.0, SXF_NOCHECKPOSITION, 0);
		SZFR P 5 A_XScream();
		SZFR Q 5 A_NoBlocking();
		SZFR RSTUV 5;
		SZFR W -1;
		stop;
	Raise:
		SZFR MLKJIH 5;
		goto See;
	}
}
