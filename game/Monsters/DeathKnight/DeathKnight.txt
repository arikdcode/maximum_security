Class Deathknight : Actor
{
//$Category Monsters/Custom/Undead
  Default
  {
    Health 1250;
    Radius 24;
    Height 72;
    Mass 700;
    Speed 10;
    PainChance 20;
    Obituary "%o thought he could fight a dark lord without his padawan at his side.";
    HitObituary "Kro-Bakul hacked %o into pieces.";
    SeeSound "monster/dknsit";
    PainSound "monster/dknpai";
    DeathSound "monster/dkndth";
    ActiveSound "monster/dknact";
    MONSTER;
    +FLOORCLIP
    +NOTARGET
    +NORADIUSDMG
    //+MISSILEMORE
	MissileChanceMult 0.5;
    +BOSS
    +DEFLECT
  }

  States
  {
  Spawn:
    DKNT AB 10 A_Look();
    Loop;
  See:
    DKNT A 0 A_Jump(32,"See2");
    DKNT A 0 A_UnSetReflectiveInvulnerable();
    DKNT AABBCCDD 3 A_Chase();
    Goto See;
  See2:
    DKNT P 0 A_SetReflectiveInvulnerable();
    DKNT PPQQRRSS 3 A_Chase();
    Loop;
  Melee:
    DKNT E 0 A_UnSetReflectiveInvulnerable();
    DKNT E 6 A_FaceTarget();
    DKNT F 1 A_Playsound ("monster/dknswg");
    DKNT F 6 A_FaceTarget();
    DKNT G 6 A_CustomMeleeAttack(10*random(1, 8), "monster/dknhit");
    Goto See;
  Missile:
    DKNT A 0 A_Jump(88,"Missile2");
    DKNT A 0 A_Jump(88,"Missile3");
    DKNT E 0 Bright A_UnSetReflectiveInvulnerable();
    DKNT E 6 Bright A_FaceTarget();
    DKNT F 6 Bright A_Playsound ("monster/kntswg");
    DKNT G 0 Bright A_SpawnProjectile("DKDart",32,0,-3,0);
    DKNT G 0 Bright A_SpawnProjectile("DKDart",32,0,-1,0);
    DKNT G 0 Bright A_SpawnProjectile("DKDart",32,0, 1,0);
    DKNT G 5 Bright A_SpawnProjectile("DKDart",32,0, 3,0);
    DKNT A 0 A_Jump(64,"Missile3");
    Goto See;
  Missile2:
    DKNT T 0 Bright A_SetReflectiveInvulnerable();
    DKNT T 6 Bright A_FaceTarget();
    DKNT U 1 Bright A_FaceTarget();
    DKNT U 0 Bright A_SpawnProjectile("FHeadShot",44,-4,-12,0);
    DKNT U 5 Bright A_SpawnProjectile("FHeadShot",44,-4, 12,0);
    DKNT U 0 Bright A_UnSetReflectiveInvulnerable();
    DKNT U 0 A_Jump(64,1);
    Goto See;
  Missile3:
    DKNT T 0 Bright A_SetReflectiveInvulnerable();
    DKNT T 3 Bright A_FaceTarget();
    DKNT U 3 Bright A_SpawnProjectile("DKbolt",44,-4,0,0);
    DKNT T 3 Bright A_FaceTarget();
    DKNT U 3 Bright A_SpawnProjectile("DKbolt",44,-4,0,0);
    DKNT T 3 Bright A_FaceTarget();
    DKNT U 3 Bright A_SpawnProjectile("DKbolt",44,-4,0,0);
    DKNT T 3 Bright A_FaceTarget();
    DKNT U 3 Bright A_SpawnProjectile("DKbolt",44,-4,0,0);
    DKNT T 3 Bright A_FaceTarget();
    DKNT U 3 Bright A_SpawnProjectile("DKbolt",44,-4,0,0);
    DKNT T 3 Bright A_FaceTarget();
    DKNT U 2 Bright A_SpawnProjectile("DKbolt",44,-4,0,0);
    DKNT U 0 Bright A_UnSetReflectiveInvulnerable();
    Goto See;
  Pain:
    DKNT H 2;
    DKNT H 2 A_Pain();
    DKNT T 105 A_SetReflectiveInvulnerable();
    DKNT T 0 A_UnSetReflectiveInvulnerable();
    DKNT P 0 A_Jump(16, "See");
    Goto See2;
  Death:
    DKNT I 0 Bright A_SpawnProjectile("DKSword", 44, 32,-90,0);
    DKNT I 8 Bright A_SpawnProjectile("DKShield",44,-32, 90,0);
    DKNT J 8 Bright A_Scream();
    DKNT K 8 Bright;
    DKNT L 8 Bright A_NoBlocking();
    DKNT MN 8 Bright;
    DKNT O -1;
    Stop;
  Raise:
    DKNT ONMLKJI 8 Bright;
    Goto See;
  }
}
